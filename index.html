# Geocoding Lnder
loca <-
  tdat %$%
  tibble(name = unique(c(reporter, partner))) %>%
  mutate(ggeo = name %>% map(. %>% googleway::google_geocode())) %>%
  mutate(x = ggeo %>% map(. %>% {.$results$geometry$location$lng}),
         y = ggeo %>% map(. %>% {.$results$geometry$location$lat})) %>%
  mutate_at(c("x", "y"),
            . %>%
              map(. %>% {ifelse((is.null(.) | length(.)!=1), NA, .)}) %>%
              unlist())

# Handelsnetzwerk
tnet <-
  tdat %>% 
  select(reporter, partner, trade_value_usd) %>% 
  rename(tval = trade_value_usd) %>% 
  igraph::graph_from_data_frame(directed = TRUE) %>% 
  as_tbl_graph() %>% 
  activate(nodes) %>% 
  left_join(loca, by = "name") %>% 
  filter(!is.na(x) & !is.na(y)) %>% 
  activate(edges) %>% 
  filter(tval > 1000)

tnet %>% 
  activate(nodes) %>% 
  {create_layout(., "manual", node.positions=as_tibble(.)[c("x","y")])} %>% 
  ggraph() +
  geom_polygon(aes(x=long, y=lat, group=group), data=wrld,
               fill="grey80", colour="grey80", size=0) +
  geom_edge_arc(aes(edge_alpha=..index.., edge_width=tval),
                edge_colour="grey20", curvature=-0.1) +
  geom_node_point(shape=21, fill="indianred1", colour="grey90") +
  scale_edge_width_continuous(range=c(0,2)) +
  scale_edge_alpha_continuous(range=c(0,0.5)) +
  coord_map(projection="mercator") +
  theme(legend.position="none",
        plot.margin=structure(c(1, 1, 1, 1), unit="lines",
                              valid.unit=3L, class="unit"),
        plot.background=element_rect(fill="grey90"),
        legend.key=element_rect(fill="grey30", colour="grey30")) +
  guides(edge_alpha="none",
         size=guide_legend(override.aes=list(size=c(3,1.5),
                                             fill=c("#b3de69", "#fdb462"))),
         fill="none",
         edge_colour=guide_legend(title="Verbindung"))
